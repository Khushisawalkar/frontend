import { AetherClient } from '@aetherflow/sdk';

// Initialize the edge client
const client = new AetherClient({
  apiKey: process.env.AETHER_API_KEY,
  region: 'global-edge'
});

// Stream telemetry data in sub-milliseconds

  userId: 'usr_948a7b',
